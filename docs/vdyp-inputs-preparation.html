3.2.3 check & fix species comp for rank1

3.2.3.1 Issues Addressed

  1. Species percentage > 0 but species code is missing → removed
  2. Total species percentage ≠ 100 → normalized
  3. Duplicate species codes within a feature → aggregated
  4. Species not ordered by dominance → reordered
  5. Invalid species codes (e.g., LS → L, VW → CW) → corrected
  6. Features with no valid species → flagged

3.2.3.2 Key Processing Steps

  • Convert wide species columns to long format for validation
  • Remove invalid or blank species codes
  • Standardize species and site index codes
  • Aggregate duplicate species by summing percentages
  • Reorder species by descending percentage
  • Reshape back to wide format (species_cd_1–6, species_pct_1–6)
  • Normalize percentages:
    • If total < 100 → add difference to species 1
    • If total > 100 → scale all species proportionally
  • Round percentages to 1 decimal place
  • Track all changes using change_log

3.2.3.3 Output

  • Table: prov_updated_spp_2024
  • Includes cleaned species composition and change tracking
library(tidyr)
library(dplyr)
library(DBI)

# -----------------------------------------------
# Pull source data
# -----------------------------------------------
query <- "
SELECT feature_id,est_site_index_species_cd,
       species_cd_1, species_pct_1,
       species_cd_2, species_pct_2,
       species_cd_3, species_pct_3,
       species_cd_4, species_pct_4,
       species_cd_5, species_pct_5,
       species_cd_6, species_pct_6
FROM vdyp7_input_layer_2024
WHERE rank_cd_update = 1;
"

spp_comp <- dbGetQuery(con, query)

# -----------------------------------------------
# Initialize change tracking field
# -----------------------------------------------
spp_comp$change_log <- NA_character_

add_change <- function(log, tag) {
  ifelse(
    is.na(log),
    tag,
    paste(log, tag, sep = " | ")
  )
}

all_features <- spp_comp %>% distinct(feature_id)
# -----------------------------------------------
# Fix outdated est_site_index_species_cd
# -----------------------------------------------
spp_comp <- spp_comp %>%
  mutate(
    original_si_cd = est_site_index_species_cd,
    est_site_index_species_cd = case_when(
      toupper(est_site_index_species_cd) == "LS" ~ "L",
      toupper(est_site_index_species_cd) == "VW" ~ "CW",
      TRUE ~ toupper(est_site_index_species_cd)
    )
  )

changed_si <- spp_comp %>%
  filter(original_si_cd != est_site_index_species_cd,
         !is.na(original_si_cd)) %>%
  distinct(feature_id)

if (nrow(changed_si) > 0) {
  spp_comp$change_log[
    spp_comp$feature_id %in% changed_si$feature_id
  ] <- add_change(
    spp_comp$change_log[
      spp_comp$feature_id %in% changed_si$feature_id
    ],
    "SITE_INDEX_SPECIES_UPDATED"
  )
}

spp_comp <- spp_comp %>% select(-original_si_cd)

# -----------------------------------------------
# Pivot to long format
# -----------------------------------------------
species_long_cd <- spp_comp %>%
  pivot_longer(
    cols = starts_with("species_cd_"),
    names_to = "rank",
    names_pattern = "species_cd_(\\d+)",
    values_to = "species_cd"
  )

species_long_pct <- spp_comp %>%
  pivot_longer(
    cols = starts_with("species_pct_"),
    names_to = "rank",
    names_pattern = "species_pct_(\\d+)",
    values_to = "percentage"
  ) %>%
  mutate(percentage = replace_na(percentage, 0))

species_long <- inner_join(
  species_long_cd,
  species_long_pct,
  by = c("feature_id", "rank")
)

# -----------------------------------------------
# Remove invalid species codes
# -----------------------------------------------
species_long <- species_long %>%
  filter(
    !is.na(species_cd),
    trimws(species_cd) != ""
  )

# -----------------------------------------------
# Detect features with NO valid species
# -----------------------------------------------
valid_ids <- species_long %>% distinct(feature_id)

no_valid_species_ids <- setdiff(
  all_features$feature_id,
  valid_ids$feature_id
)

if (length(no_valid_species_ids) > 0) {
  spp_comp$change_log[
    spp_comp$feature_id %in% no_valid_species_ids
  ] <- add_change(
    spp_comp$change_log[
      spp_comp$feature_id %in% no_valid_species_ids
    ],
    "NO_VALID_SPECIES"
  )
}

# -----------------------------------------------
# Fix outdated species codes
# -----------------------------------------------
species_long <- species_long %>%
  mutate(
    original_cd = species_cd,
    species_cd = case_when(
      toupper(species_cd) == "LS" ~ "L",
      toupper(species_cd) == "VW" ~ "CW",
      TRUE ~ toupper(species_cd)
    )
  )

changed_codes <- species_long %>%
  filter(original_cd != species_cd) %>%
  distinct(feature_id)

if (nrow(changed_codes) > 0) {
  spp_comp$change_log[
    spp_comp$feature_id %in% changed_codes$feature_id
  ] <- add_change(
    spp_comp$change_log[
      spp_comp$feature_id %in% changed_codes$feature_id
    ],
    "SPECIES_CODE_UPDATED"
  )
}

species_long <- species_long %>% select(-original_cd)

# -----------------------------------------------
# Aggregate duplicate species
# -----------------------------------------------
dup_check <- species_long %>%
  group_by(feature_id, species_cd) %>%
  tally() %>%
  filter(n > 1)

if (nrow(dup_check) > 0) {
  spp_comp$change_log[
    spp_comp$feature_id %in% dup_check$feature_id
  ] <- add_change(
    spp_comp$change_log[
      spp_comp$feature_id %in% dup_check$feature_id
    ],
    "DUPLICATE_SPECIES_AGGREGATED"
  )
}

aggregated <- species_long %>%
  group_by(feature_id, species_cd) %>%
  summarise(total_percentage = sum(percentage), .groups = "drop")


# -----------------------------------------------
# Order species by descending percentage
# -----------------------------------------------
aggregated <- aggregated %>%
  group_by(feature_id) %>%
  arrange(desc(total_percentage), .by_group = TRUE) %>%
  mutate(order = row_number()) %>%
  ungroup()

# -----------------------------------------------
# Spread back to wide
# -----------------------------------------------
df_species <- aggregated %>%
  select(feature_id, order, species_cd) %>%
  pivot_wider(
    names_from = order,
    values_from = species_cd,
    names_prefix = "species_cd_",
    values_fill = NA
  )

df_percentage <- aggregated %>%
  select(feature_id, order, total_percentage) %>%
  pivot_wider(
    names_from = order,
    values_from = total_percentage,
    names_prefix = "species_pct_",
    values_fill = 0
  )

df_merged <- all_features %>%
  left_join(df_species, by = "feature_id") %>%
  left_join(df_percentage, by = "feature_id")
# -----------------------------------------------
# Ensure pct = 0 when species_cd is NA
# -----------------------------------------------
pct_cols <- grep("^species_pct_\\d+$", names(df_merged), value = TRUE)
cd_cols  <- gsub("pct", "cd", pct_cols)

for (i in seq_along(pct_cols)) {
  df_merged[[pct_cols[i]]][is.na(df_merged[[cd_cols[i]]])] <- 0
}

# -----------------------------------------------
# Normalization
# -----------------------------------------------
df_merged$sum_pct <- rowSums(df_merged[pct_cols], na.rm = TRUE)

# Under 100
idx <- df_merged$sum_pct < 100 & df_merged$sum_pct > 0
if (any(idx)) {
  diff_needed <- 100 - df_merged$sum_pct[idx]
  df_merged$species_pct_1[idx] <- df_merged$species_pct_1[idx] + diff_needed
  
  spp_comp$change_log[
    spp_comp$feature_id %in% df_merged$feature_id[idx]
  ] <- add_change(
    spp_comp$change_log[
      spp_comp$feature_id %in% df_merged$feature_id[idx]
    ],
    "UNDER100_INCREASED_SPP1"
  )
}

# Over 100
idx <- df_merged$sum_pct > 100
if (any(idx)) {
  scale_factor <- 100 / df_merged$sum_pct[idx]
  df_merged[idx, pct_cols] <- df_merged[idx, pct_cols] * scale_factor
  
  spp_comp$change_log[
    spp_comp$feature_id %in% df_merged$feature_id[idx]
  ] <- add_change(
    spp_comp$change_log[
      spp_comp$feature_id %in% df_merged$feature_id[idx]
    ],
    "OVER100_SCALED_DOWN"
  )
}

# -----------------------------------------------
# Round to 1 decimal
# -----------------------------------------------
df_merged[pct_cols] <- round(df_merged[pct_cols], 1)

# -----------------------------------------------
# Merge change log back
# -----------------------------------------------
updated_spp <- df_merged %>%
  left_join(
    spp_comp %>% select(feature_id, change_log),
    by = "feature_id"
  )

# -----------------------------------------------
# If no changes occurred
# -----------------------------------------------
updated_spp$change_log[is.na(updated_spp$change_log)] <- "NO_ADJUSTMENT"

# -----------------------------------------------
# Final QA Check
# -----------------------------------------------
updated_spp$sum_check <- rowSums(updated_spp[pct_cols], na.rm = TRUE)

issues <- updated_spp %>%
  filter(abs(sum_check - 100) > 0.2 & sum_check != 0)

if (nrow(issues) > 0) {
  warning("Some rows deviate from 100% by > 0.2 — inspect.")
  print(issues)
}

# -----------------------------------------------
# Write to database
# -----------------------------------------------
dbWriteTable(
  con,
  "prov_updated_spp_2024",
  updated_spp,
  row.names = FALSE,
  overwrite = TRUE
)

3.2.4 Update Age & Height

  1. Missing age/height for species 2:
    • If species_pct_2 > 0 and age/height is missing → copy from species 1
  2. Invalid ordering (species 2 > species 1):
    • If est_age_spp2 > est_age_spp1 → reset to species 1
    • If est_height_spp2 > est_height_spp1 → reset to species 1
  3. Invalid height values:
    • If est_height_spp2 == -1 → set to 0

3.2.4.1 Output

  • Table: prov_updated_ba_age_ht_2024
  • All updates are tracked in change_log

       is.na(layer_attri$est_height_spp2) &

       !is.na(layer_attri$est_height_spp1) &
       layer_attri$est_height_spp2 > layer_attri$est_height_spp1

layer_attri$est_height_spp2[idx] <- layer_attri$est_height_spp1[idx]
layer_attri$change_log[idx] <- add_change(layer_attri$change_log[idx], "HT2_gt_HT1_fixed")

3.2.5 review Basal Area, Age & Height

3.2.5.1 Checks Performed

  1. Missing layer identifiers:
    • layer_level_code is NULL
    • vdyp7_layer_cd is NULL
  2. Basal Area issues:
    • basal_area_75 is NULL or < 1
  3. Age & Height issues:
    • est_age_spp1 or est_height_spp1 is NULL
    • est_age_spp1 > 350 (old stands, may not have yields generated)
    • species_pct_2 > 0 but age/height missing
  4. density issues:
    • stems_per_ha_75 is NULL or < 1
  5. Crown closure issues:
    • crown_closure is NULL or < 1

dbWriteTable(con, "prov_ba_age_ht_checks_2024", layer_attri, overwrite = TRUE, row.names = FALSE)

3.2.6 Update Other-Veg comp

  1. Missing Data Handling:
    • Replace all numeric NA values with 0
    • Flag missing BEC zone codes
  2. Vegetation Cover Adjustment:
    • If shrub_crown_closure + herb_cover_pct + bryoid_cover_pct > 100: → calculate scaling ratio = 100 / total → apply proportional reduction using floor() → log as “VEG_SCALED_DOWN”
  3. Non-Vegetation Cover Adjustment:
    • If sum of non_veg_cover_pct_1–3 > 100: → scale proportionally using same method → log as “NONVEG_SCALED_DOWN”
  4. Land Cover Adjustment:
    • If sum of land_cover_pct_1–3 > 100: → scale proportionally → log as “LANDCOVER_SCALED_DOWN”
  5. Ratio Tracking:
    • Store applied scaling ratios for:
      • vegetation (ratio_veg)
      • non-vegetation (ratio_non_veg)
      • land cover (ratio_land)
  6. Change Logging:
    • All adjustments tracked in change_log
    • Records with no changes labeled as “NO_ADJUSTMENT”

# Check missing BEC zone
